% --- Cấu hình ngôn ngữ ---
\usepackage[vietnamese]{babel}
\usepackage{fontspec}

% --- Chọn font chữ ---
% Sử dụng Extension=.ttf để tránh lỗi font với XeLaTeX
\setmainfont{CrimsonPro}[
    Extension=.ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    BoldItalicFont=*-BoldItalic,
    Path=/home/kiet/.local/share/fonts/Crimson_Pro/static/
]

\setsansfont{OpenSans}[
    Extension=.ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    BoldItalicFont=*-BoldItalic,
    Path=/home/kiet/.local/share/fonts/Open_Sans/static/
]

% Nếu không có CaskaydiaCove, dùng FreeMono làm fallback
\setmonofont{FreeMono}    % Font đơn cách (monospace)

% --- Điều chỉnh kích thước font ---
% Giảm 20% so với kích thước mặc định để phù hợp với khổ A5
\usepackage{anyfontsize}

\renewcommand{\normalsize}{\fontsize{9}{10.8}\selectfont}
\renewcommand{\small}{\fontsize{8}{9.6}\selectfont}
\renewcommand{\footnotesize}{\fontsize{7}{8.4}\selectfont}
\renewcommand{\large}{\fontsize{10}{12}\selectfont}
\renewcommand{\Large}{\fontsize{11}{13.2}\selectfont}
\renewcommand{\LARGE}{\fontsize{12}{14.4}\selectfont}
\renewcommand{\huge}{\fontsize{14}{16.8}\selectfont}
\renewcommand{\Huge}{\fontsize{16}{19.2}\selectfont}
